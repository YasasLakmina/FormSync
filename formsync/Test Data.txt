$schema: "http://json-schema.org/draft-07/schema#"
type: object
properties:
  fullName:
    type: string

  email:
    type: string

  notifications:
    type: boolean

  address:
    type: object
    properties:
      city:
        type: string
      postalCode:
        type: string
    required:
      - city

required:
  - fullName
